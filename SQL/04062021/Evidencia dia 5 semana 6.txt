create database TrazabilidadProduccion character set utf8;

use TrazabilidadProduccion;

create table TablaTrazabilidadProduccion(
id_variedad int auto_increment,
codigoProductor varchar(20),
FechaProduccion date,
tipoEmbalaje text,
primary key(id_variedad)
);
alter table TablaTrazabilidadProduccion add column CodigoPallet integer;
alter table TablaTrazabilidadProduccion add column FechaEmbalaje date;
alter table TablaTrazabilidadProduccion modify codigoProductor int;
alter table TablaTrazabilidadProduccion drop column FechaProduccion;

create database TrazabilidadBodega character set utf8;

use TrazabilidadBodega;

create table TablaTrazabilidadbodega(
id_CodigoPallet int auto_increment,
codigoProductor varchar(20),
FechaIngreso date,
tipoEmbalaje text,
tipoEmbarque text,
primary key(id_CodigoPallet)
);
alter table TablaTrazabilidadbodega add column Temperatura float;
alter table TablaTrazabilidadbodega add column PesoPallet float;
alter table TablaTrazabilidadbodega modify PesoPallet int;
alter table TablaTrazabilidadbodega drop column tipoEmbalaje;
